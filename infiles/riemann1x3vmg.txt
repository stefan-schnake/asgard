-title    : Riemann 1x3v
-subtitle : mixed grid, quadratic basis

# df / dt + v d_x f = \nu C(f)
# C(f) := d_v( (v-u)f + th d_v f )
-pde : riemann_1x3v

# disctretization in space
# mixed grid with level 7 full grid in x 
# and level 6 sparse grid in v.
# quadratic basis
-grid         : mixed

# dt  = 0.32*CFL
# T   = dt*250 = 0.05
-step-method : imex
